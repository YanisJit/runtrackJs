// Fonction utilitaire pour vérifier si un nombre est premier function estPremier(num) { if (num <= 1) return false; for (let i = 2; i <= Math.sqrt(num); i++) { if (num % i === 0) return false; } return true; } function sommeNombresPremiers(a, b) { if (estPremier(a) && estPremier(b)) { return a + b; } else { return false; } } // Test console.log(sommeNombresPremiers(3, 5)); // Affiche 8
